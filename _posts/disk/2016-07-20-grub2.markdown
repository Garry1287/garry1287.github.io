# grub2

Установка grub2
mount /dev/vda2 /mnt
mount --bind /dev /mnt/dev
mount --bind /proc /mnt/proc
chroot /mnt
export PS1="(chroot) $PS1"
grub2-install /dev/vda



cacti:/mnt # umount /mnt/sys
cacti:/mnt # umount /mnt/dev
cacti:/mnt # umount /mnt/sys








grub2-mkconfig -o /boot/grub2/grub.cfg
